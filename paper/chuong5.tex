\chapter{ĐÁNH GIÁ KẾT QUẢ VÀ KẾ HOẠCH TIẾP THEO}

Sau khi hoàn thành các bước cấu hình hệ thống WooCommerce và tích hợp các tính năng bổ trợ trên website \textbf{Sen Việt Tea}, dự án đã bước vào giai đoạn vận hành thử nghiệm (testing phase). Chương này sẽ tập trung đánh giá các kết quả thực tế đạt được thông qua giao diện demo trực quan, báo cáo tiến độ chi tiết, đo lường các chỉ số kỹ thuật ban đầu về hiệu năng tải trang và mức độ tối ưu hóa cho công cụ tìm kiếm (SEO). Đồng thời, kế hoạch triển khai Digital Marketing cùng giải pháp bảo mật và sao lưu dữ liệu trong giai đoạn cuối kỳ cũng sẽ được hoạch định một cách cụ thể.

\textbf{Phân tích kết quả kỹ thuật:}
\begin{itemize}
    \item \textbf{Trên máy tính để bàn (Desktop):} Điểm hiệu năng đạt \textbf{93/100} nhờ việc tối giản mã nguồn theme, sử dụng các truy vấn WooCommerce gọn nhẹ và không tải các thư viện dư thừa. Các chỉ số quan trọng (Core Web Vitals) như LCP (1.4 giây) và CLS (0.02) đều nằm trong vùng xanh an toàn của Google.
    \item \textbf{Trên thiết bị di động (Mobile):} Điểm hiệu năng dừng lại ở mức trung bình khá \textbf{78/100}. Nguyên nhân chính do tốc độ xử lý phần cứng giả lập trên thiết bị di động chậm hơn và việc một số hình ảnh biểu ngữ (banner) chưa được nén tối ưu sang định dạng thế hệ mới (như WebP). 
    \item \textbf{Khuyến nghị tối ưu hóa:} Cần kích hoạt nén ảnh WebP tự động, trì hoãn tải các file JavaScript chưa cần thiết (Defer JS) và nén dung lượng các file CSS để giảm thời gian phản hồi FCP trên di động xuống dưới 2.0 giây.
\end{itemize}

\section{Kế hoạch giai đoạn cuối kỳ}

\subsection{Kế hoạch Digital Marketing}

Sau khi website chính thức đi vào hoạt động ổn định trên môi trường internet thực tế, để thu hút khách hàng và gia tăng doanh số bán trà trực tuyến, kế hoạch Digital Marketing sẽ được triển khai đồng bộ qua ba kênh chính:

\subsubsection{Chiến lược tối ưu hóa công cụ tìm kiếm (SEO)}
\begin{itemize}
    \item \textbf{Nghiên cứu từ khóa (Keyword Research):} Tập trung vào ba nhóm từ khóa chính:
    \begin{itemize}
        \item Từ khóa thương hiệu: "Sen Việt Tea", "trà Sen Việt".
        \item Từ khóa sản phẩm chính: "trà thái nguyên ngon", "trà sen hồ tây cao cấp", "mua ấm tử sa dáng tây thi".
        \item Từ khóa thông tin (hỗ trợ traffic gián tiếp): "cách pha trà oolong ngon", "tác dụng của trà xanh đối với sức khỏe".
    \end{itemize}
    \item \textbf{Xây dựng nội dung (Content Marketing):} Đều đặn xuất bản 2 bài viết chất lượng cao mỗi tuần trên trang kiến thức trà để thu hút khách hàng tiềm năng tự nhiên.
    \item \textbf{SEO Off-page:} Tiến hành tìm kiếm các liên kết chất lượng (Backlinks) từ các trang blog chuyên về ẩm thực, các diễn đàn thưởng trà uy tín tại Việt Nam.
\end{itemize}

\subsubsection{Chiến dịch quảng cáo tìm kiếm (Google Ads)}
\begin{itemize}
    \item \textbf{Google Search Ads:} Thiết lập quảng cáo văn bản nhắm mục tiêu chuẩn xác khi người dùng tìm kiếm các từ khóa có mục đích mua hàng cao như "quà tặng trà tết", "hộp quà trà cao cấp".
    \item \textbf{Google Shopping:} Liên kết kho hàng WooCommerce với Google Merchant Center để hiển thị trực quan sản phẩm trà kèm giá bán ngay phía trên kết quả tìm kiếm của khách hàng.
\end{itemize}

\subsubsection{Chiến dịch tiếp thị mạng xã hội (Facebook Ads)}
\begin{itemize}
    \item \textbf{Nhắm mục tiêu đối tượng (Audience Targeting):} Tiếp cận tệp khách hàng có sở thích liên quan đến uống trà, thiền định, gốm nghệ thuật, đồ cổ và phân khúc khách hàng mua quà tặng doanh nghiệp trung và cao cấp.
    \item \textbf{Tiếp thị lại (Remarketing):} Cài đặt mã Facebook Pixel trên website nhằm chạy các chiến dịch bám đuổi các khách hàng đã xem trang sản phẩm hoặc đã thêm sản phẩm vào giỏ hàng nhưng chưa nhấn thanh toán, kích thích họ quay lại hoàn tất đơn hàng.
\end{itemize}

\subsection{Kế hoạch bảo mật và sao lưu dữ liệu (Backup)}

An toàn thông tin và tính toàn vẹn của cơ sở dữ liệu là yếu tố quan trọng hàng đầu trong vận hành cửa hàng trực tuyến nhằm bảo vệ quyền lợi của khách hàng lẫn uy tín thương hiệu.

\subsubsection{Kế hoạch sao lưu dữ liệu định kỳ (Backup Strategy)}
\begin{itemize}
    \item \textbf{Công cụ thực hiện:} Sử dụng plugin chuyên dụng \textbf{UpdraftPlus} kết hợp với tệp cấu hình PHP đã tối ưu hóa giới hạn bộ nhớ lớn (2GB) để đảm bảo quá trình xuất file diễn ra trơn tru.
    \item \textbf{Tần suất sao lưu:}
    \begin{itemize}
        \item Cơ sở dữ liệu (Database): Sao lưu tự động hàng ngày (\textit{Daily backup}) vào lúc 0:00 sáng để lưu giữ toàn bộ dữ liệu đơn hàng và khách hàng mới.
        \item Mã nguồn và tệp tin đa phương tiện (Plugins, Themes, Uploads): Sao lưu tự động hàng tuần (\textit{Weekly backup}) do mã nguồn ít khi có sự biến động lớn.
    \end{itemize}
    \item \textbf{Không gian lưu trữ ngoại biên:} Cấu hình tự động đẩy các bản sao lưu lên tài khoản lưu trữ đám mây Google Drive của doanh nghiệp thay vì lưu trữ trên cùng một máy chủ vật lý, tránh rủi ro mất mát dữ liệu toàn cục khi xảy ra lỗi phần cứng.
\end{itemize}

\subsubsection{Cấu hình bảo mật hệ thống (Security Strategy)}
\begin{itemize}
    \item \textbf{Chứng chỉ bảo mật SSL/HTTPS:} Bắt buộc mã hóa toàn bộ dữ liệu đường truyền để bảo mật tuyệt đối thông tin cá nhân và tài khoản thanh toán của khách hàng khi giao dịch qua cổng PayPal.
    \item \textbf{Chống tấn công thử sai (Brute-force protection):} Kích hoạt plugin \textbf{Limit Login Attempts Reloaded} thiết lập quy tắc tự động khóa địa chỉ IP truy cập trong vòng 24 giờ nếu nhập sai mật khẩu trang quản trị quá 5 lần.
    \item \textbf{Kiểm soát Spam:} Sử dụng dịch vụ \textbf{Akismet Anti-Spam} để tự động lọc và loại bỏ các bình luận rác mang liên kết độc hại trên trang kiến thức trà và đánh giá sản phẩm.
    \item \textbf{Bảo trì định kỳ:} Lên lịch cập nhật phiên bản WordPress Core, WooCommerce và các plugin đi kèm tối thiểu 1 lần/tháng để kịp thời vá các lỗ hổng bảo mật mới phát sinh.
\end{itemize}
