% --- 6. CẤU HÌNH ĐỀ MỤC ---
\usepackage{titlesec}
\usepackage{needspace} % Đảm bảo tiêu đề không bị tách khỏi nội dung

% Cho phép đánh số đến cấp thứ 3 (subsubsection)
\setcounter{secnumdepth}{3}

% Định nghĩa lại cách đánh số: 1., 1.1., 1.1.1.
\renewcommand{\thesection}{\arabic{section}.}
\renewcommand{\thesubsection}{\thesection\arabic{subsection}.}
\renewcommand{\thesubsubsection}{\thesubsection\arabic{subsubsection}.}

% TÊN CHƯƠNG: 14pt, CHỮ IN HOA, ĐẬM, Giữa
\titleformat{\chapter}[display]
  {\normalfont\fontsize{14}{16}\bfseries\centering\selectfont}
  {\MakeUppercase{\chaptertitlename\ \thechapter}}
  {0pt}{\MakeUppercase}
\titlespacing*{\chapter}{0pt}{-20pt}{20pt}

% TIỂU MỤC CẤP 1: 13pt, CHỮ IN HOA, ĐẬM, Đều (Justified)
% needspace đảm bảo ít nhất 5 dòng sau tiêu đề cùng trang
\titleformat{\section}
  {\needspace{5\baselineskip}\normalfont\fontsize{13}{15}\bfseries\selectfont}
  {\thesection}{1em}{\MakeUppercase}

% Tiểu mục cấp 2: 13pt, Chữ thường, đậm, Đều
\titleformat{\subsection}
  {\needspace{4\baselineskip}\normalfont\fontsize{13}{15}\bfseries\selectfont}
  {\thesubsection}{1em}{}

% Tiểu mục cấp 3: 13pt, Chữ thường, đậm, nghiêng, Đều
\titleformat{\subsubsection}
  {\needspace{3\baselineskip}\normalfont\fontsize{13}{15}\bfseries\itshape\selectfont}
  {\thesubsubsection}{1em}{}
